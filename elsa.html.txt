<h3>❄️戴冠式エルサ練習モード</h3>
<canvas id="game" width="360" height="540"></canvas><br>
<button onclick="skill()">スキル発動</button>
<div id="score">Score:0</div>

<script>
const c=document.getElementById("game");
const ctx=c.getContext("2d");
let score=0,ice=false;

function draw(){
ctx.fillStyle=ice?"#8de0ff":"#2c3f8f";
ctx.fillRect(0,0,c.width,c.height);
ctx.fillStyle="white";
ctx.fillText(ice?"氷ループ中❄️":"通常",150,50);
requestAnimationFrame(draw);
}
draw();

c.addEventListener("click",()=>{
if(ice){
score+=500000;
}else{
score+=100000;
}
document.getElementById("score").textContent="Score:"+score;
});

function skill(){
ice=true;
setTimeout(()=>ice=false,5000);
}
</script>
